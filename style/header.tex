% --- Per-chapter ink colour ---
% Odd chapters (Units / application) = PURPLE
% Even chapters (Deep-Dives)          = GREEN
% Body text also shifts to the chapter ink.
\newcommand{\chapterink}{%
  \ifodd\value{chapter}\color{darkpurple}\else\color{deepgreen}\fi}
\newcommand{\chapterinkaccent}{%
  \ifodd\value{chapter}\color{midpurple}\else\color{accentgreen}\fi}

% Suppress "Chapter N" prefix entirely
\renewcommand{\chaptername}{}

% Chapter title colour is injected by the Lua filter directly into the heading inlines.
% \titleformat just sets font size/weight; no colour here.
\titleformat{\chapter}[block]
  {\LARGE\bfseries}
  {}{0pt}
  {}
\titlespacing*{\chapter}{0pt}{-20pt}{20pt}
% Section/subsection headings inherit the current text colour
% (set by the Lua filter after each H1)
\titleformat{\section}
  {\Large\bfseries}
  {\thesection}{1em}
  {}
\titleformat{\subsection}
  {\large\bfseries}
  {\thesubsection}{1em}
  {}
\titleformat{\subsubsection}
  {\normalsize\bfseries}
  {\thesubsubsection}{1em}
  {}

% === Block quotes: colour follows chapter ink ===
\usepackage{mdframed}
\usepackage{etoolbox}

% Dynamically resolve quote colours from the current chapter counter.
% \chapterquotecolor expands at mdframed-begin time, not at definition time.
% Odd (Unit) = purple tones, Even (Deep-Dive) = green tones
\newcommand{\chapterquoteborder}{\ifodd\value{chapter}midpurple\else accentgreen\fi}
\newcommand{\chapterquotebg}{\ifodd\value{chapter}lightpurple\else lightgreen\fi}
\newcommand{\chapterquotetext}{\ifodd\value{chapter}darkpurple\else deepgreen\fi}

\AtBeginEnvironment{quote}{%
  \edef\cqborder{\chapterquoteborder}%
  \edef\cqbg{\chapterquotebg}%
  \edef\cqtext{\chapterquotetext}%
  \begin{mdframed}[
    linecolor=\cqborder,
    linewidth=3pt,
    leftline=true,
    rightline=false,
    topline=false,
    bottomline=false,
    backgroundcolor=\cqbg,
    innerleftmargin=12pt,
    innerrightmargin=12pt,
    innertopmargin=8pt,
    innerbottommargin=8pt,
    skipabove=12pt,
    skipbelow=12pt,
  ]
  \itshape\color{\cqtext}
}
\AtEndEnvironment{quote}{\end{mdframed}}
